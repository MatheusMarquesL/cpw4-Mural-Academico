 function handleNovoAviso(dados: Omit<Aviso, "id" | "data" | "favorito">) {
    const data = new Date().toLocaleDateString("pt-BR");
    setAvisos((prev) => [
      { id: Date.now().toString(), ...dados, data, favorito: false },
      ...prev,
    ]);
  }

  function handleFavoritar(id: string) {
    setAvisos((prev) =>
      prev.map((a) => (a.id === id ? { ...a, favorito: !a.favorito } : a)),
    );
    setAvisoDetalhe((prev) =>
      prev?.id === id ? { ...prev, favorito: !prev.favorito } : prev,
    );
  }

  function handleExcluir(id: string) {
    setAvisos((prev) => prev.filter((a) => a.id !== id));
  }

  function handleLimparDados() {
    if (confirm("Tem certeza que deseja limpar todos os avisos?"))
      setAvisos([]);
  }

   const avisosFiltrados = useMemo(
